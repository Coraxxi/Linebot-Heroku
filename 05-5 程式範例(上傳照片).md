# 05-5 程式範例(上傳照片)

### CLI命令:

```
1. 在[D槽]建立一個<myApp>資料夾.
------------------------------------------------ 
d:
cd\
md myApp
cd myApp


2. 建立一個本地論的git
------------------------------------------------ 
git init


3. 產生package.json
------------------------------------------------ 
npm init


4. 登入heroku
------------------------------------------------ 
heroku login


5. 登入git
------------------------------------------------ 
若未安裝git cli, 先下載安裝:https://git-scm.com/downloads


6. 建立一個heroku專案
** 如果已有專業, 跳過本步驟, 執行下個步驟
------------------------------------------------ 
heroku create myTestApp
              +-------+ (請自訂名稱)
              
              
7. 切換至某個已存在的heroku專案
** 如果執行5建立專案, 跳過本步驟
------------------------------------------------ 
heroku git:remote -a myTestApp
                     +-------+ (自己的專案名稱)             


8. 修改package.json, 增加index.js(程式如下)
------------------------------------------------ 


9. 安裝模組
------------------------------------------------ 
npm install
npm install express --save
npm install linebot --save
npm install @line/bot-sdk --save
npm install pg --save


10. 上傳本地端git至heroku專案的git空間
------------------------------------------------ 
git add .
git commit -am "myApp"
git push heroku master


11. 查看Heroku的Console畫面
------------------------------------------------ 
heroku logs --tail
```


### (1) 增加index.js, 修改package.json

```
 <myApp>
   |___ <.git>
   |___ <node_modules>
   |
   |___ <public>
   |        |___ <imgs>
   |               |___ (存放上傳照片的位置, 先建空資料夾)   
   |
   |___ index.js
   |___ package.json  
```



### (2) 資料表名稱:message, 其建立SQL如下:

```
-- -----------------------------------------
-- serno, 流水號 (自動遞增)
-- uid,   使用者的id
-- name,  使用者姓名
-- msg,   使用者傳來的文字訊息
-- img,   使用者傳來的圖檔名稱,
--         圖檔存在<public/imgs>中.
-- -----------------------------------------
DROP TABLE IF EXISTS "public"."message";
CREATE TABLE "public"."message" (
    "serno" int8 NOT NULL DEFAULT nextval('message_serno_seq'::regclass),
    "uid" varchar(255) COLLATE "pg_catalog"."default",
    "name" varchar(255) COLLATE "pg_catalog"."default",
    "msg" varchar(255) COLLATE "pg_catalog"."default",
    "img" varchar(255) COLLATE "pg_catalog"."default"
)
; 
```

### (4) index.js

```javascript
//--------------------------------
// 載入必要的模組
//--------------------------------
var linebot = require('linebot');
var express = require('express');
var fs = require('fs');
const { Client } = require('pg');
const line = require('@line/bot-sdk');


//--------------------------------
// 填入自己在linebot的channel值
//--------------------------------
var cId = '(填入自己的channelId資料)';
var cSecret = '(填入自己的channelSecret資料)';
var cToken = '(填入自己的channelAccessToken資料)';
var pgConn = 'postgres://(填入自己的postgres URI資料)';
var webURL = 'https://(填入自己網站名稱).herokuapp.com/imgs/';

var bot = linebot({
    channelId: cId,
    channelSecret: cSecret,
    channelAccessToken: cToken
});


//--------------------------------
// 機器人接受訊息的處理
//--------------------------------
bot.on('message', function(event) {
    //取得使用者訊息
    event.source.profile().then(
        function (profile) {		
        //如果使用者上傳訊息的型態是圖片
        if(event.message.type=='image'){
            event.message.content().then(function (content) {
            //建立一個可以取回使用者傳到line中圖片的client					
            const client = new line.Client({
                channelAccessToken: cToken
            });

            //由client向line取圖片(圖片編號是event.message.id)
            var chunks = [];  
            var size = 0;  

            client.getMessageContent(event.message.id)
                .then((stream) => {
                    //圖片切成幾塊分別下載
                    stream.on('data', (chunk) => {
                        chunks.push(chunk);  
                        size += chunk.length;  
                    });
						
                    //如果發生下載錯誤
                    stream.on('error', (err) => {
                        console.log('取回圖片錯誤');
                        return;
                    });
						
                    //全部下載完成
                    stream.on('end', (err) => {
                        var data = null;  
                        switch(chunks.length) {  
                            case 0: data = new Buffer(0);  
                                break;  
                            case 1: data = chunks[0];  
                                break;  
                            default:
                                //將切分圖片組合									
                                data = new Buffer(size);  
                                for (var i = 0, pos = 0, l = chunks.length; i < l; i++) {  
                                    var chunk = chunks[i];  
                                    chunk.copy(data, pos);  
                                    pos += chunk.length;  
                                } 

                                var fileName = event.message.id + '.jpg';
                                console.log('圖片組合完成, 檔名預設為:' + fileName);					  
										
                                //將圖片寫入檔案
                                fs.writeFile('./public/imgs/' + fileName, data, (err) => {
                                    if (err) throw err;											
                                    console.log('已存圖檔');

                                    //建立資料庫連線           
                                    var client = new Client({
                                        connectionString: pgConn,
                                        ssl: true,
                                    })
											
                                    client.connect();
											
                                    //待寫出的資料
                                    console.log(profile.userId);
                                    console.log(profile.displayName);
                                    console.log(event.message.id);
											
                                    //寫出資料(資料庫欄位名稱不使用駝峰命名, 否則可能出錯)
                                    client.query("insert into message(uid, name, img) values ($1, $2, $3)", [profile.userId, profile.displayName, event.message.id ], (err, results) => {    	
                                    //寫出結果
                                    if (err){
                                        console.log('DB失敗');
                                    }else{						
                                        console.log('已存DB');
                                    }

                                    //關閉連線
                                    client.end();	
												
                                    //回覆使用者上傳的圖片
                                    return event.reply({
                                        type: 'image',
                                        originalContentUrl: webURL + fileName,
                                        previewImageUrl: webURL + fileName
                                    });											  
                                });										  							
                            });
										
                        break;  
                    }  	  
                });	
            });				
        }).catch(function(error) {
            console.log('無法取出使用者上傳圖片' + error);
            return;
        });				
    }else{
        return event.reply('姓名:' + profile.displayName + ' 編號:' + profile.userId + ' 訊息:' +  event.message.text)
            .then(function (data) {
                console.log('已回傳訊息');
            }).catch(function (error) {
                console.log('回傳訊息失敗');
            });
        }	
    });
});


//--------------------------------
// 建立一個網站應用程式app,
// 如果連接根目錄, 交給機器人處理,
// 在line的webHook中設定.
//--------------------------------
const app = express();
const linebotParser = bot.parser();
app.post('/', linebotParser);


//--------------------------------
// 設定網站可直接取用檔案的資料夾
//--------------------------------
app.use(express.static('public'));


//--------------------------------
// 監聽3000埠號, 
// 或是監聽Heroku設定的埠號
//--------------------------------
var server = app.listen(process.env.PORT || 3000, function() {
    var port = server.address().port;
    console.log("正在監聽埠號:", port);
});
```
