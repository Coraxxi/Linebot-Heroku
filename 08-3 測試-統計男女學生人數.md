## 1. asyncDB.js 
```
填入自己的pgConn
```
#### 程式
``` js
'use strict';

//-----------------------
// 引用資料庫模組
//-----------------------
const {Client} = require('pg');

//-----------------------
// 自己的資料庫連結位址
//-----------------------
var pgConn = 'postgres://自己的資料庫連結URI';


//產生可同步執行query物件的函式
function query(sql, value=null) {
    return new Promise((resolve, reject) => {
        //設定資料庫連線物件
        var client = new Client({
            connectionString: pgConn,
            ssl: true
        })     

        //連結資料庫
        client.connect();

        //回覆查詢結果  
        client.query(sql, value, (err, results) => {                   
            if (err){
                reject(err);
            }else{
                resolve(results);
            }

            //關閉連線
            client.end();
        });
    });
}

//匯出
module.exports = query;
```


## 2. student.js 
```
這個程式提供以下API讓前端呼叫:

1. 統計男/女學生人數
   (1)countStudents() 
   (2)傳入值: 無
   (3)回傳值:
        |__ -9(執行錯誤)
        |__ {male:男生人數, female:女生人數}        
```
#### 程式
``` js
'use strict';

//引用操作資料庫的物件
const query = require('./asyncDB');

//------------------------------------------
// 統計男/女學生人數
//------------------------------------------
var countStudents = async function(){
    //存放結果
    let result={};  

    //計算男生
    await query("select count(*) as cnt from student where gender = 'M'")
        .then((data) => {
            if(data.rows.length > 0){
                result.male = data.rows[0].cnt;  //男生人數
            }else{
                result.male = 0;  //找不到資料
            }    
        }, (error) => {
            result = -9;  //執行錯誤
        });

    //計算女生
    await query("select count(*) as cnt from student where gender = 'F'")
        .then((data) => {
            if(data.rows.length > 0){
                result.female = data.rows[0].cnt;  //男生人數
            }else{
                result.female = 0;  //找不到資料
            }    
        }, (error) => {
            result = -9;  //執行錯誤
        });     

    //回傳執行結果
    return result;  
}
//------------------------------------------

//匯出
module.exports = {countStudents};
```
